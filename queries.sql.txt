-- 1. High Value Transactions
SELECT * 
FROM transactions
WHERE amount > 50000;
-- 2. Multiple Transactions in Same Day
SELECT customer_id, COUNT(*) AS txn_count
FROM transactions
GROUP BY customer_id, DATE(transaction_date)
HAVING COUNT(*) > 2;
-- 3. Previous Transaction (Window Function)
SELECT 
    customer_id,
    transaction_date,
    amount,
    LAG(transaction_date) OVER (PARTITION BY customer_id ORDER BY transaction_date) AS prev_txn
FROM transactions;
-- 4. Rapid Transactions (within 5 minutes)
WITH txn_diff AS (
    SELECT 
        customer_id,
        transaction_date,
        LAG(transaction_date) OVER (PARTITION BY customer_id ORDER BY transaction_date) AS prev_txn
    FROM transactions
)
SELECT *
FROM txn_diff
WHERE TIMESTAMPDIFF(MINUTE, prev_txn, transaction_date) < 5;
-- 5. Location Change Detection
SELECT 
    customer_id,
    location,
    LAG(location) OVER (PARTITION BY customer_id ORDER BY transaction_date) AS prev_location
FROM transactions;
-- 6. Fraud Score Calculation
SELECT 
    customer_id,
    SUM(
        CASE 
            WHEN amount > 50000 THEN 1
            WHEN transaction_type = 'international' THEN 1
            ELSE 0
        END
    ) AS fraud_score
FROM transactions
GROUP BY customer_id
ORDER BY fraud_score DESC;
